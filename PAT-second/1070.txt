#include<stdlib.h>
int main(){
	int a[10000];
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	//ÔËÓÃÃ°ÅÝÅÅÐò
	int temp;
	for(int i=0;i<n;i++){
		for(int j=i;j<n;j++){
			if(a[i]>a[j]){
				temp=a[i];
				a[i]=a[j];
				a[j]=temp;
			}
		}
	}

		sum+=a[i];
		sum=sum/2;
		
	}

}
